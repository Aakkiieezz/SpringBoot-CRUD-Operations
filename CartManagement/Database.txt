Creating table is optional, since spring hibernate automatically creates if not present
Due to mode (spring.jpa.hibernate.ddl-auto=update) in application.properties

CREATE TABLE `cart_items` (
  `id` int NOT NULL AUTO_INCREMENT,
  `date` varchar(255) DEFAULT NULL,
  `name` varchar(255) DEFAULT NULL,
  `price` int NOT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=10 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;
SELECT * FROM CartDatabase.cart_items;
